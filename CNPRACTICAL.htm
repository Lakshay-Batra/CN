Program 5 

Aim:Write a Program to make a Telnet Client

Description: A Telnet Client connect to the remote hostsIt can send and receive data to and from remote host

Language:Java

Program:

import java.io.*;

import java.net.*;

import java.util.*;

public class telnetClient

{

            public static void main(String[] args)throws Exception

            {         

                        BufferedReader in=new BufferedReader(new InputStreamReader(System.in));

                        Socket s=null;

                        boolean x=true;                       

                        while(x)

                        {

                                    System.out.print("telnetClient>");

                                    String r=in.readLine();

                                    StringTokenizer t1=new StringTokenizer(r);

                                    String req=t1.nextToken();

                                    if(req.equals("quit"))

                                    System.exit(0);

                                    send:if(req.equals("send"))

                                    {

                                                if(s==null)

                                                {

                                                            System.out.println("not connected");

                                                            break send;

                                                }

                                                OutputStream out=s.getOutputStream();

                                                String data=t1.nextToken();

                                                while(data !=null)

                                                {

                                                            byte buf[]=data.getBytes();

                                                            out.write(buf);

                                                            data=t1.nextToken();

                                                }

                                                out.close();

                                    }

                                    display:if(req.equals("display"))

                                    {

                                                if(s==null)

                                                {

                                                            System.out.println("not connected");

                                                            break display;

                                                }

                                                System.out.println(s.getRemoteSocketAddress());

                                    }

                                    if(req.equals("connect"))

                                    {

                                                String hostName=t1.nextToken();

                                                int port=Integer.parseInt(t1.nextToken());

                                                if(s!=null)

                                                s.close();

                                                s=new Socket(InetAddress.getByName(hostName),port);

                                                System.out.println("Connected");

                                    }                     

                        }

            }

}

import java.io.*;

import java.net.*;

import java.util.*;

public class ftpClient

{

            public static void main(String[] args)throws Exception

            {

                        BufferedReader ins=new BufferedReader(new InputStreamReader(System.in));

                        Socket s=null;

                        boolean x=true;                       

                        while(x)

                        {

                                    System.out.print("ftpClient>");

                                    String r=ins.readLine();

                                    StringTokenizer t1=new StringTokenizer(r);

                                    String req=t1.nextToken();

                                    if(req.equals("quit"))

                                    System.exit(0);

                                    if(req.equals("open"))

                                    {

                                                if(s!=null)

                                                s.close();

                                                String hostName=t1.nextToken();

                                                int port=Integer.parseInt(t1.nextToken());

                                                if(s!=null)

                                                s.close();

                                                s=new Socket(InetAddress.getByName(hostName),port);

                                                System.out.println("Connected");

                                    }

                                    recev:if(req.equals("recev"))

                                    {

                                                if(s==null)

                                                {

                                                            System.out.println("not connected");

                                                            break recev;

                                                }

 

                                                InputStream in=s.getInputStream();

                                                String filename=t1.nextToken();

                                                BufferedOutputStream fout=new BufferedOutputStream(new FileOutputStream(filename));

                                                int c;

                                                while((c=in.read())!=-1)

                                                {

                                                            fout.write((char)c);

                                                }

                                                System.out.println("Download complete");

                                                fout.close();

                                                in.close();

                                    }

                                    send:if(req.equals("send"))

                                    {

                                                if(s==null)

                                                {

                                                            System.out.println("not connected");

                                                            break send;

                                                }

 

                                                OutputStream out=s.getOutputStream();

                                                String filename=t1.nextToken();

                                                BufferedInputStream fin=new BufferedInputStream(new FileInputStream(filename));

                                                int c;

                                                while((c=fin.read())!=-1)

                                                {

                                                            out.write((char)c);

                                                }

                                                System.out.println("Download complete");

                                                fin.close();

                                                out.close();

                                    }

                        }

            }

}
